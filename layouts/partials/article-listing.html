{{/* Reusable article listing row. Context: either a Page (default — no icons), or a dict like (dict "page" $page "showIcons" true "popularPaths" $paths) where $paths is a slice of RelPermalink strings for the currently popular pages. When showIcons is true, a Featured icon (star) is rendered for .Params.featured pages; otherwise a Popular icon (trending-up) is rendered if the page's RelPermalink is in popularPaths. Featured wins when both apply. */}} {{- $page := . -}} {{- $showIcons := false -}} {{- $popularPaths := slice -}} {{- if reflect.IsMap . -}} {{- $page = .page -}} {{- $showIcons = .showIcons | default false -}} {{- $popularPaths = .popularPaths | default slice -}} {{- end -}} {{- $dateFormat := site.Params.dateFormat | default "Jan 2, 2006" -}} {{- $isFeatured := and $showIcons (eq $page.Params.featured true) -}} {{- $isPopular := and $showIcons (not $isFeatured) (in $popularPaths $page.RelPermalink) -}}
{{- if $isFeatured -}}{{ partial "icons/featured.html" }}{{- end -}} {{- if $isPopular -}}{{ partial "icons/popular.html" }}{{- end -}} {{ $page.Title }} {{ $page.Section }} — {{ $page.PublishDate.Format $dateFormat }}
{{ if $page.Params.summary }} {{ $page.Params.summary | markdownify | truncate 200 }} {{ end }}
